TABLE USERS   

ID INTEGER PRIMARY KEY AUTOINCREMENT,
   FIRST_NAME VARCHAR(40) NOT NULL,
   LAST_NAME VARCHAR(40) NOT NULL,
   AGE INTEGER NOT NULL,
   ADDRESS VARCHAR(70),
   CITY VARCHAR(40),
   STATE VARCHAR(40),
   COUNTRY VARCHAR(40),
   POSTALCODE VARCHAR(10),
   PHONE VARCHAR(24) NOT NULL UNIQUE,
   EMAIL TEXT NOT NULL UNIQUE,
   PASSWORD CHAR(50)
